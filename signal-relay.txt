BIND w north s south a west d east

CMD north
 MATCH1
  CALL move_north

CMD south
 MATCH1
  CALL move_south

CMD west
 MATCH1
  CALL move_west

CMD east
 MATCH1
  CALL move_east

CMD haveA
 MATCH1
  P P

CMD haveB
 MATCH1
  Q Q

CMD haveC
 MATCH1
  R R

CMD haveF
 MATCH1
  N N

CMD preC
 MATCH1
  r r

CMD setA
 TRY_ALL
  p P

CMD setB
 TRY_ALL
  q Q

CMD setC
 TRY_ALL
  r R overload

CMD setF
 TRY_ALL
  n N

CMD overload
 TRY_ALL
  m x

CMD move_north
 MATCH1
  FOR s -m
   s 1
   1 -

  a 1 setA
  1 -

  b 1 haveA! setB
  1 -

  f 1 haveB! setF
  1 -

  c 1 haveF! setC
  1 -

  d 1 haveB!
  1 -

  u 1 preC!
  1 -

  h 1 haveC!
  1 -

  e E haveC!
  1 -

CMD move_south
 MATCH1
  FOR s -m
   1 -
   s 1

  1 -
  a 1 setA

  1 -
  b 1 haveA! setB

  1 -
  f 1 haveB! setF

  1 -
  c 1 haveF! setC

  1 -
  d 1 haveB!

  1 -
  u 1 preC!

  1 -
  h 1 haveC!

  1 -
  e E haveC!

CMD move_west
 MATCH1
  FOR s -m
   s1 1s

  a1 1- setA

  b1 1- haveA! setB

  f1 1- haveB! setF

  c1 1- haveF! setC

  d1 1- haveB!

  u1 1- preC!

  h1 1- haveC!

  e1 E- haveC!

CMD move_east
 MATCH1
  FOR s -m
   1s s1

  1a -1 setA

  1b -1 haveA! setB

  1f -1 haveB! setF

  1c -1 haveF! setC

  1d -1 haveB!

  1u -1 preC!

  1h -1 haveC!

  1e -E haveC!
