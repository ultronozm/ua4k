  1- -1

  2- -2

 ;; winning the game
 ATOMIC
  1a -A

  2b -B

  ;; using keys to open gates
  ATOMIC
   pk -p

   qg -q

  ;; key pickup without unlock
  ATOMIC
   pk -p

   q- -q

  ;; bumping into walls, possibly picking up keys
  FOR s -k
   ATOMIC
    ps -p
 
    qx qx
