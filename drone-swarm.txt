BIND w north s south a west d east z release_locked

CMD north
 ATOMIC
  CALL_EACH breach_north move1_north move2_north sync_locks_north

CMD south
 ATOMIC
  CALL_EACH breach_south move1_south move2_south sync_locks_south

CMD west
 ATOMIC
  CALL_EACH breach_west move1_west move2_west sync_locks_west

CMD east
 ATOMIC
  CALL_EACH breach_east move1_east move2_east sync_locks_east

CMD release_locked
 TRY_ALL
  A a
  - 1

  - 1
  A a

  B b
  - 2

  - 2
  B b

CMD both_locked
 ATOMIC

  A A

  B B

  CALL park_unlock

CMD release_if_any_locked
 ATOMIC

  MATCH1

   A A

   B B

  CALL release_locked

CMD sync_locks_north
 MATCH1

  CALL both_locked

  CALL bounce_back_north

  ? ?

CMD sync_locks_south
 MATCH1

  CALL both_locked

  CALL bounce_back_south

  ? ?

CMD sync_locks_west
 MATCH1

  CALL both_locked

  CALL bounce_back_west

  ? ?

CMD sync_locks_east
 MATCH1

  CALL both_locked

  CALL bounce_back_east

  ? ?

  m x
